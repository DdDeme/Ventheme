function hexToRgb(hex) {
const r = parseInt(hex.slice(1,3),16);
const g = parseInt(hex.slice(3,5),16);
const b = parseInt(hex.slice(5,7),16);
return `${r},${g},${b}`;
}
function hexToRgba(hex, alpha=1) {
return `rgba(${hexToRgb(hex)},${alpha})`;
}
function darken(hex, amount) {
let r = parseInt(hex.slice(1,3),16);
let g = parseInt(hex.slice(3,5),16);
let b = parseInt(hex.slice(5,7),16);
r = Math.max(0, Math.floor(r*(1-amount)));
g = Math.max(0, Math.floor(g*(1-amount)));
b = Math.max(0, Math.floor(b*(1-amount)));
return `rgb(${r},${g},${b})`;
}

// ── CSS GENERATOR ──
function generateCSS(v) {
const lines = [
`/**`,
` * VenTheme Export`,
` * Generated: ${new Date().toLocaleDateString()}`,
` * Import into: Vencord > Settings > Themes`,
` */`,
``,
`:root {`,
` --background-primary: ${v.bg};`,
` --background-secondary: ${v.sidebar};`,
` --background-secondary-alt: ${darken(v.sidebar, 0.15)};`,
` --background-tertiary: ${darken(v.sidebar, 0.25)};`,
` --background-accent: ${v.chanActive};`,
` --background-floating: ${v.inputBg};`,
` --background-mobile-primary: ${v.bg};`,
` --background-mobile-secondary: ${v.sidebar};`,
` --channeltextarea-background: ${v.inputBg};`,
` --text-normal: ${v.text};`,
` --text-muted: ${hexToRgba(v.text, 0.5)};`,
` --text-link: ${v.accent};`,
` --brand-experiment: ${v.accent};`,
` --brand-experiment-560: ${v.accent};`,
` --interactive-normal: ${hexToRgba(v.text, 0.7)};`,
` --interactive-hover: ${v.text};`,
` --interactive-active: ${v.text};`,
`}`,
``,
`/* ── Typography ── */`,
`* { font-family: ${v.fontFamily} !important; }`,
`.theme-dark { font-size: ${v.fontSize}px; line-height: ${v.lineHeight}; }`,
``,
`/* ── Border Radius ── */`,
`.container_bd5d48, .sidebar_ded4b5 { border-radius: ${v.borderRadius}px; }`,
`.avatar_b37765 { border-radius: ${v.avatarRadius}% !important; }`,
`.message_d5deea { border-radius: ${Math.min(parseInt(v.borderRadius),8)}px; }`,
``,
`/* ── Sidebar Width ── */`,
`.sidebar_ded4b5 { width: ${v.sidebarWidth}px !important; }`,
];

if (v.glass) {
lines.push(``, `/* ── Glassmorphism ── */`);
lines.push(`.sidebar_ded4b5 { background: rgba(${hexToRgb(v.sidebar)},${v.sidebarOpacity}) !important; backdrop-filter: blur(20px) !important; }`);
lines.push(`.panels_a4d4d9 { background: rgba(${hexToRgb(v.sidebar)},${v.sidebarOpacity*0.9}) !important; backdrop-filter: blur(20px) !important; }`);
} else if (v.sidebarOpacity < 1) {
lines.push(``, `/* ── Sidebar Opacity ── */`);
lines.push(`.sidebar_ded4b5 { opacity: ${v.sidebarOpacity} !important; }`);
}

if (v.bgImg && v.bgUrl) {
lines.push(``, `/* ── Background Image ── */`);
lines.push(`.chat_a7d72e { background: url('${v.bgUrl}') center/cover !important; }`);
if (parseInt(v.bgBlur) > 0) {
lines.push(`.chat_a7d72e::before { content: ''; position: absolute; inset: 0; backdrop-filter: blur(${v.bgBlur}px); z-index: 0; }`);
}
}

if (v.compact) {
lines.push(``, `/* ── Compact Mode ── */`);
lines.push(`.message_d5deea { margin: 2px 0 !important; padding: 2px 16px !important; }`);
lines.push(`.avatar_b37765 { display: none !important; }`);
}

const css = lines.join('\n');
document.getElementById('css-output').value = css;
return css;
}

// ── MODAL ──
function openModal() {
const css = document.getElementById('css-output').value;
document.getElementById('modal-css-preview').textContent = css;
document.getElementById('modal').classList.add('open');
}
function closeModal() {
document.getElementById('modal').classList.remove('open');
}

// ── COPY ──
function copyCSS() {
const css = document.getElementById('css-output').value;
navigator.clipboard.writeText(css).then(() => showToast('✓', 'CSS copied to clipboard!'));
}

// ── DOWNLOAD ──
function downloadCSS() {
const css = document.getElementById('css-output').value;
const blob = new Blob([css], { type: 'text/css' });
const a = document.createElement('a');
a.href = URL.createObjectURL(blob);
a.download = 'ventheme.css';
a.click();
URL.revokeObjectURL(a.href);
showToast('↓', 'Downloaded ventheme.css!');
closeModal();
}

// ── TOAST ──
function showToast(icon, msg) {
const t = document.getElementById('toast');
document.getElementById('toast-icon').textContent = icon;
document.getElementById('toast-msg').textContent = msg;
t.classList.add('show');
setTimeout(() => t.classList.remove('show'), 2500);
}
